13       public static void main(String [] args ) {
14           SComposer c = new SComposer();

16          System.out.println("c.mms1 ..."); c.mms1(); space(c);
17          System.out.println("c.mms2 ..."); c.mms1(); space(c);
18          System.out.println("c.mms3 ..."); c.mms1(); space(c);
19          System.out.println("c.mms4 ..."); c.mms1(); space(c);
20          System.out.println("c.mms5 ..."); c.mms1(); space(c);
21          System.out.println("c.mms6 ..."); c.mms1(); space(c);
22          System.out.println("c.mms7 ..."); c.mms1(); space(c);
23          System.out.println("c.mms8 ..."); c.mms1(); space(c);
24          c.untext();
25       }
26   
27       private static void space(SComposer c) {
28   
29           c.untext(); c.rest(2); c.text();
30       }
