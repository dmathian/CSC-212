8    public class Interpreter3 {
9        private void interpreter() {
10           // CREATE OBJECTS HERE
11           SPainter miro = new SPainter("Dot Thing", 400, 400);
12           miro.setScreenLocation(0, 0);
13           SCircle dot = new SCircle(180);
14   
15           // REPEAT A COMMAND FROM AN INPUT DIALOG BOX TO INTERPRET IT
16           while (true) {
17               String command = JOptionPane.showInputDialog(null, "Command?");
18               if (command == null) {
19                   command = "exit";
20               } //user clicked
21               if (command.equalsIgnoreCase("blue")) {
22                   miro.setColor(Color.BLUE);
23                   miro.paint(dot);
24               } else if (command.equalsIgnoreCase("red")) {
25                   miro.setColor(Color.RED);
26                   miro.paint(dot);
27               } else if (command.equalsIgnoreCase("green")) {
28                   miro.setColor(Color.GREEN);
29                   miro.paint(dot);
30               } else if (command.equalsIgnoreCase("yellow")) {
31                   miro.setColor(Color.YELLOW);
32                   miro.paint(dot);
33               } else if (command.equalsIgnoreCase("random")) {
34                   miro.setColor(randomColor());
35                   miro.paint(dot);
36               } else if (command.equalsIgnoreCase("help")) {
37                   JOptionPane.showMessageDialog(null, "valid commans are: "
38                           + "RED | BLUE | GREEN | YELLOW | RANDOM | HELP | EXIT ");
39               } else if (command.equalsIgnoreCase("exit")) {
40                   miro.end();
41                   System.out.println("Thank you for viewing dots ...");
42                   break;
43               } else {
44                   JOptionPane.showMessageDialog(null, "Unrecongnizable command: "
45                           + command.toUpperCase());
46               }
47           }
48       }
49

51           interpreter();

60       private static Color randomColor() {
61           int rv = (int) (Math.random()*256);
62           int gv = (int) (Math.random()*256);
63           int bv = (int) (Math.random()*256);
64           return new Color(rv,gv,bv);
65       }
