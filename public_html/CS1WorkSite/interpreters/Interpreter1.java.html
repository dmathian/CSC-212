1    package interpreters;

4    /* 
5     *This interpreter is inteded to paint differnt colored dots in a window. 
6     * 
7     * The commands that the interpreter can recongnize and respond to are: 
8     * -Blue: paint a blue dot 
9     * -Red: paont a red dot 
10    * -Help: show a list of the commands in a lialog message box. 
11    * -Exit: terminate the program 
12    */
13   
14   import painter.SPainter;
15   import shapes.SCircle;
16   import javax.swing.*;
17   import java.awt.*;
18   
19   
20   public class Interpreter1 {
21   
22       private void interpreter() {
23   
24           // CREATE OBJECTS TO THINK WITH
25           SPainter miro = new SPainter("Dot Thing", 400, 400);
26           miro.setScreenLocation(0, 0);
27           SCircle dot = new SCircle(180);
28   
29           // REPEATEDLY TAKE A COMMAND FROM AN IN PUT DIALOG BOX AND INTERPRET IT
30           while (true) {
31   
32               String command = JOptionPane.showInputDialog(null, "command?");
33               if (command == null) {
34                   command = "exit";
35               } // user clicked on cancel
36               if (command.equalsIgnoreCase("yellow")) {
37                   miro.setColor(Color.YELLOW);
38                   miro.paint(dot);
39   
40               } if (command.equalsIgnoreCase("blue")) {
41                   miro.setColor(Color.BLUE);
42                   miro.paint(dot);
43   
44               } if (command.equalsIgnoreCase("red")) {
45                   miro.setColor(Color.RED);
46                   miro.paint(dot);
47   
48               } else if (command.equalsIgnoreCase("help")) {
49   
50                   JOptionPane.showMessageDialog(null, "Valid commands are: "
51   
52                           + "RED | BLUE | HELP | EXIT ");
53               } else if (command.equalsIgnoreCase("exit")) {
54   
55                   miro.end();
56                   System.out.println("Thank you for viewing the dots...");
57   
58   
59               } else {
60   
61                   JOptionPane.showMessageDialog(null, "Unrecognizable command:"
62                           + command.toUpperCase());
63               }
64   
65           }
66       }
67          //INFRASTRUCTURE FOR SOME SIMPLE PAINTING
68   
69       public Interpreter1() {
70           interpreter();
71       }
72       public  static void main(String [] args ){
73   
74           SwingUtilities.invokeLater(new Runnable() {
75               @Override
76               public void run() {
77                   new Interpreter1();
78               }
79           });
80   
81          }
82   
83       }
