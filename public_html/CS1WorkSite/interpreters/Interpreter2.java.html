20       private void interpreter() {
21

27           // REPEATEDLY TAKE A COMMAND FROM AN IN PUT DIALOG BOX AND INTERPRET IT WHILE (TRUE) {
28   
29           String command = JOptionPane.showInputDialog(null, "command?");
30   
31           if (command == null) {command = "exit"; } // user clicked on cancel
32           if (command.equalsIgnoreCase ("blue"))
33           if (command.equalsIgnoreCase("yellow"))
34           if (command.equalsIgnoreCase("green"))
35           if (command.equalsIgnoreCase("red")){
36   
37               miro.setColor(Color.BLUE);
38               miro.setColor(Color.YELLOW);
39               miro.setColor(Color.GREEN);
40               miro.setColor(Color.RED);
41               miro.paint(dot);
42   
43   
44           } else if (command.equalsIgnoreCase("help")) {
45   
46               JOptionPane.showMessageDialog(null, "Valid commands are: "
47   
48                       + "GREEN | YELLOW | RED | BLUE | HELP | EXIT ");
49           } else if (command.equalsIgnoreCase("exit")) {
50   
51               miro.end();
52               System.out.println("Thank you for viewing the dots...");
53   
54   
55           } else {
56   
57               JOptionPane.showMessageDialog(null, "Unrecognizable command:"
58                       + command.toUpperCase());
59           }
60   
61       }
