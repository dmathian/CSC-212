26       private void paintTheImage() {
27           SPainter painter = new SPainter("Alternate Balloons", 300, 300);
28           paintSky(painter); // ask IntelliJ to generate the stub
29           int nrOfAlternateBalloons = 100;
30         paintBalloons(painter,nrOfAlternateBalloons); //ask IntelliJ to generate the stub
31       }
32       private void paintSky(SPainter painter){
33           painter.setColor(Color.BLUE);
34           SSquare sky = new SSquare(2000);
35           painter.paint(sky);
36       }
37       private void paintBalloons(SPainter painter, int nrOfAlternateBalloons){
38           int i =1;
39           while (i <= nrOfAlternateBalloons){
40              paintOneBalloon(painter); // ask IntelliJ to generate the stub
41               i = 1 + 1;
42           }
43       }
44       private void paintOneBalloon(SPainter painter) {
45           Random rgen = new Random();
46           int rn = rgen.nextInt(3);
47           if (rn == 0) {
48               painter.setColor(Color.RED);
49   
50           } else if (rn == 1 ) {
51               painter.setColor(Color.WHITE);
52   
53           } else {
54               painter.setColor(Color.BLUE);
55           }
56           painter.move();
57           int balloonRadius = 30;
58           SCircle balloon = new SCircle(balloonRadius);
59           painter.paint(balloon);
60           painter.setColor(Color.BLACK);
61           painter.draw(balloon);
62   
63       }
