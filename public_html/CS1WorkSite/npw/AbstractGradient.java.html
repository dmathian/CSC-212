12   public class AbstractGradient {

14       public static void main(String[] args) {
15          invokeLater( AbstractGradient::new );

18           private AbstractGradient() {
19                      paintTheImage();
20                  }
21   
22        private void paintTheImage(){
23                      // Grab the input information
24                  int width = getNumber("width");
25                  int height = getNumber("height");
26                  int colWidth = getNumber("column width");
27                      // Establish the painter
28                      SPainter painter = new SPainter("Colorful Abstract Gradient", width, height);
29                      SCircle dot = new SCircle(5);
30   
31                      // Move the painter to the upper left corner to get ready to paint the gradient
32                      painter.mfd(height/2);
33                      painter.tl(90);
34                      painter.mfd(width/2 - 10);
35                      painter.tl(90);
36   
37                      // Paint it!
38                      paintGradient(painter, dot, height, width, colWidth);
39                  }
40   
41           private static int getNumber(String prompt) {
42                      String nss = JOptionPane.showInputDialog(null,prompt+"?");
43                      Scanner scanner = new Scanner(nss);
44                      return scanner.nextInt();
45                  }
46   
47           private void paintGradient(SPainter painter, SCircle dot, int height, int width, int colWidth){
48                   int cols = 0;
49   
50                      // Calculate the number of columns.
51                      // any dots only half on the canvas, so we subtract some space.
52                 int nrOfCols = ( width / colWidth ) - 2;
53                  while (cols < nrOfCols){
54                              nextCol(painter, colWidth);
55                              paintColumn(painter, dot, height);
56                              cols = cols + 1;
57                    }
58                  }
59   
60           private void paintOneDot(SPainter painter, SCircle dot){
61                   painter.paint(dot);
62                  }
63   
64                   // Dots are spaced tighter.
65           private void paintColumn(SPainter painter, SCircle dot, int height) {
66                      int travel = 0;
67                      int totalDistanceTraveled = 0;
68   
69                      // Paint a row with decreasing distance between dots.
70                      while(totalDistanceTraveled < height - (dot.radius() * 3)) {
71                              travel = randomDistance((height - totalDistanceTraveled) / 4);
72                              totalDistanceTraveled = totalDistanceTraveled + travel;
73                              painter.mfd(travel);
74                              paintOneDot(painter, dot);
75                  }
76   
77                      // Make the method invariant with respect to painter position.
78                      painter.mbk(totalDistanceTraveled);
79                  }
80                       // Moves the painter to the next column.
81          private void nextCol(SPainter painter, int colWidth){
82                      painter.tl(90);
83                      painter.mfd(colWidth);
84                      painter.tr(90);
85                  }
86          private int randomDistance(int maxDistance){
87                      Random rgen = new Random();
88                      return rgen.nextInt(maxDistance);
89                  }
