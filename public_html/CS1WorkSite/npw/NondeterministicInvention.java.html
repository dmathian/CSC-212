26       private void paintTheImage() {
27           SPainter painter = new SPainter("Invention 2", 600, 600);
28           int nrOfRectangles = 300;
29           paintRectangles(painter, nrOfRectangles);  // ask IntelliJ to generate the stub
30       }
31   
32       private void paintRectangles(SPainter painter, int nrOfRectangles) {
33           int i = 1;
34           while (i <= nrOfRectangles) {
35               paintOneRectangle(painter);  // ask IntelliJ to generate the stub
36               i = i + 1;
37           }
38       }
39   
40       private void paintOneRectangle(SPainter painter) {
41           Random rgen = new Random();
42           int rn = rgen.nextInt(6);
43           if (rn == 0) {
44               painter.setColor(new Color(255, 83, 13 ));
45           }
46           else if (rn == 1) {
47               painter.setColor(new Color(41, 164, 255 ));
48           }
49           else if (rn == 2) {
50               painter.setColor(new Color(255, 13, 255 ));
51           }
52           else if (rn == 3) {
53               painter.setColor(new Color(28, 255, 230 ));
54           }
55           else if (rn == 4) {
56               painter.setColor(new Color(255, 250, 88 ));
57           }
58           else {
59               int r = rgen.nextInt(255);
60               int b = rgen.nextInt(255);
61               int y = rgen.nextInt(255);
62               painter.setColor(new Color(r,b,y));
63           }
64           painter.move();
65           int recSide = 30;
66           SRectangle rec = new SRectangle(recSide, recSide*2);
67           painter.paint(rec);
68           painter.setColor(Color.BLACK);
69           painter.draw(rec);
70       }
