28                  private void paintTheImage(){
29                      // Establish the painter
30                      SPainter painter = new SPainter("SimpleDots", width, height);
31                      SCircle dot = new SCircle(5);
32   
33                      // Move the painter to the upper left corner to get ready to paint the gradient
34                      painter.mfd(height/2);
35                          painter.tl(90);
36                      painter.mfd(width/2 - 10);
37                      painter.tl(90);
38   
39                      // Paint it!
40                      paintGradient(painter, dot, height, width, spacingFactor);
41                  }

43                  private static int getNumber(String prompt) {
44                      String nss = JOptionPane.showInputDialog(null,prompt+"?");
45                      Scanner scanner = new Scanner(nss);
46                      return scanner.nextInt();
47                  }
48   
49                  private static String getString(String prompt) {
50                      String nss = JOptionPane.showInputDialog(null,prompt+"?");
51                      Scanner scanner = new Scanner(nss);
52                      return scanner.next();
53                  }

55                  // Supports floating point numbers as spacingFactor values
56                  private void paintGradient(SPainter painter, SCircle dot, int height, int width, int verticalSpacing){
57                      // Calculate the number of columns. We want to fill the screen, but we don't want
58                      // any dots only half on the canvas, so we subtract some space.
59                      int nrOfCols = ( width / verticalSpacing ) - 2;
60                      int columnCount = 0;
61                      while (columnCount < nrOfCols){
62                              nextCol(painter, verticalSpacing);
63                              columnCount = columnCount + 1;
64                              paintColumn(painter, dot, height);
65   
66                          }
67                  }
68          private void paintOneDot(SPainter painter, SCircle dot){
69                      dynamicColor(painter);
70                      painter.paint(dot);
71                  }
72   
73                  private void paintColumn(SPainter painter, SCircle dot, int height) {
74                      int horizontalSpacing = spacingFactor;
75                      int displacement = 0;
76   
77                      while(displacement < height) {
78                              displacement = displacement + horizontalSpacing;
79                              painter.mfd(horizontalSpacing);
80                              paintOneDot(painter, dot);
81                          }
82                          // Make the method invariant with respect to painter position.
83                          painter.mbk(displacement);
84                      }
85   
86                  // Moves the painter to the next column.
87                  private void nextCol(SPainter painter, double colWidth){
88                      painter.tl(90);
89                      painter.mfd(colWidth);
90                      painter.tr(90);
91                  }
92          private void dynamicColor(SPainter painter) {
93                      Random rgen = new Random();
94                      if (colorChose.equalsIgnoreCase( "red" )) {
95                              painter.setColor( Color.RED );
96                          } else if (colorChose.equalsIgnoreCase( "green" )) {
97                             painter.setColor( Color.GREEN );
98                         } else if (colorChose.equalsIgnoreCase( "blue" )) {
99                             painter.setColor( Color.BLUE );
100                        } else if (colorChose.equalsIgnoreCase( "random" )) {
101  
102                      } else {
103                            painter.setColor( Color.BLACK );
104                        }
105                }
