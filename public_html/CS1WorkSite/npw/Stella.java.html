38       private void drawSquare(SPainter painter, int nrOfSquares, int shrinkFactor, SSquare square) {
39                      boolean colorSwitchFlag = true;
40                      Color randomColor1 = randomColor();
41                      Color randomColor2 = randomColor();
42   
43                      for (int i = 0; i < nrOfSquares; i=1+1) {
44                              if (colorSwitchFlag == true) {
45                                      painter.setColor( randomColor1 );
46                                      painter.paint( square );
47                                      square.shrink( shrinkFactor );
48                                      colorSwitchFlag = false;
49                                  } else {
50                                      painter.setColor( randomColor2 );
51                                      painter.paint( square );
52                                      square.shrink( shrinkFactor );
53                                      colorSwitchFlag = true;
54                                  }
55                          }
56                  }

59                      int r = rgen.nextInt( 255 );
60                      int g = rgen.nextInt( 255 );

62                      return new Color( r, g, b );
